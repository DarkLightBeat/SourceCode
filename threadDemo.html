public class ThreadDemo {
    public static void main(String[] args) throws InterruptedException {
       
        Thread numbersThread = new Thread(new PrintNumbers());
        Thread lettersThread = new Thread(new PrintLetters());
        Thread timeThread = new Thread(new PrintTime());

        
        numbersThread.start();
        lettersThread.start();
        timeThread.start();

        numbersThread.join();
        lettersThread.join();
        timeThread.join();

        System.out.println("\nAll threads have finished. Main program exiting.");
    }
}

class PrintNumbers implements Runnable {
    public void run() {
        for (int i = 1; i <= 5; i++) {
            System.out.println("Number: " + i);
            Utils.sleep(500);
        }
    }
}

class PrintLetters implements Runnable {
    public void run() {
        for (char c = 'A'; c <= 'E'; c++) {
            System.out.println("Letter: " + c);
            Utils.sleep(700);
        }
    }
}

class PrintTime implements Runnable {
    public void run() {
        for (int i = 0; i < 3; i++) {
            System.out.println("Current time: " + java.time.LocalTime.now());
            Utils.sleep(1000);
        }
    }
}

class Utils {
    static void sleep(int ms) {
        try {
            Thread.sleep(ms);
        } catch (InterruptedException e) {
            // Ignore interruption for this mini-lab
        }
    }
}
