Candy's PseudoCode


 def minCandies(ratings):
    n = len(ratings)
    if n == 0:
        return 0  # No children, no candies needed

    candies = [1] * n  # Every child gets at least 1 candy

    # Left to Right Pass
    for i in range(1, n):
        if ratings[i] > ratings[i - 1]:
            candies[i] = candies[i - 1] + 1

    # Right to Left Pass
    for i in range(n - 2, -1, -1):
        if ratings[i] > ratings[i + 1]:
            candies[i] = max(candies[i], candies[i + 1] + 1)

    return sum(candies)  # Return total minimum candies needed

# **User Input Section**
ratings = list(map(int, input("Enter ratings separated by spaces: ").split()))

# **Calling the function and printing the result**
result = minCandies(ratings)
print(f"Minimum candies needed: {result}")
