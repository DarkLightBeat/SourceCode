Race Car PseudoCode


 from collections import deque

def racecar(target):
    queue = deque([(0, 1, 0)])  # (position, speed, steps)
    visited = set((0, 1))  # Store visited (position, speed)

    while queue:
        position, speed, steps = queue.popleft()

        # If we reach the target, return the number of steps
        if position == target:
            return steps

        # Move forward with 'A' (accelerate)
        new_position = position + speed
        new_speed = speed * 2
        if (new_position, new_speed) not in visited and 0 <= new_position <= 2 * target:
            queue.append((new_position, new_speed, steps + 1))
            visited.add((new_position, new_speed))

        # Reverse direction with 'R' (reverse)
        new_speed = -1 if speed > 0 else 1
        if (position, new_speed) not in visited:
            queue.append((position, new_speed, steps + 1))
            visited.add((position, new_speed))

# **User Input Section**
try:
    target = int(input("Enter the target position: "))
    if target < 0:
        print("Target position must be a positive number.")
    else:
        result = racecar(target)
        print(f"Minimum steps needed to reach position {target}: {result}")
except ValueError:
    print("Invalid input. Please enter a valid number.")
